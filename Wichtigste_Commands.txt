# Run from your repo root (the folder that contains "backend\")
$Root = (Get-Location).Path
$OutDir = Join-Path $Root "out"
New-Item -ItemType Directory -Force -Path $OutDir | Out-Null

# What to include (adjust as needed)
$IncludeExt = @("*.py","*.ini","*.yml","*.yaml","*.toml","*.env","requirements*.txt","docker-compose*.yml")

# What to exclude (important: drop .venv and caches)
$ExcludeDirRegex = '\\(\.venv|__pycache__|\.git|node_modules|dist|build|\.pytest_cache|\.mypy_cache)\b'

# Collect files
$Files = Get-ChildItem -Path $Root -Recurse -File -ErrorAction SilentlyContinue |
  Where-Object { $_.FullName -notmatch $ExcludeDirRegex } |
  Where-Object {
    $name = $_.Name
    foreach ($pat in $IncludeExt) { if ($name -like $pat) { return $true } }
    return $false
  } |
  Sort-Object FullName

# Manifest (paths + hashes) so an AI can reason about completeness
$Manifest = $Files | ForEach-Object {
  $rel = Resolve-Path -LiteralPath $_.FullName | ForEach-Object { $_.Path.Substring($Root.Length).TrimStart('\') }
  [PSCustomObject]@{
    path = $rel
    bytes = $_.Length
    lastWriteTime = $_.LastWriteTimeUtc.ToString("o")
    sha256 = (Get-FileHash -Algorithm SHA256 -LiteralPath $_.FullName).Hash
  }
}
$Manifest | ConvertTo-Json -Depth 4 | Set-Content -Encoding UTF8 (Join-Path $OutDir "manifest.json")

# Chunked concatenation
$MaxChunkBytes = 250KB   # tune for your AI input limits
$chunkIndex = 1
$chunkPath = Join-Path $OutDir ("chunk_{0:D4}.txt" -f $chunkIndex)
$chunkBytes = 0

"" | Set-Content -Encoding UTF8 $chunkPath

foreach ($f in $Files) {
  $rel = Resolve-Path -LiteralPath $f.FullName | ForEach-Object { $_.Path.Substring($Root.Length).TrimStart('\') }

  $header = "`n`n===== FILE: $rel =====`n"
  $content = Get-Content -LiteralPath $f.FullName -Raw -ErrorAction SilentlyContinue

  # Estimate bytes (UTF8)
  $block = $header + $content
  $blockBytes = [System.Text.Encoding]::UTF8.GetByteCount($block)

  if (($chunkBytes + $blockBytes) -gt $MaxChunkBytes -and $chunkBytes -gt 0) {
    $chunkIndex++
    $chunkPath = Join-Path $OutDir ("chunk_{0:D4}.txt" -f $chunkIndex)
    "" | Set-Content -Encoding UTF8 $chunkPath
    $chunkBytes = 0
  }

  Add-Content -Encoding UTF8 -LiteralPath $chunkPath -Value $block
  $chunkBytes += $blockBytes
}

"Done. Wrote manifest.json and $chunkIndex chunk file(s) to $OutDir"
